## 더 맵게 No.42626

```java
import java.util.*;

class Solution {
    public int solution(int[] scoville, int K) {

        /*
            매번 최소값 2개를 찾아야 하기 때문에 PriorityQueue 사용
            우선순위큐는 정렬되어 저장됨
        */
        PriorityQueue<Integer> queue = new PriorityQueue<>();
        int count = 0;         // 섞는 횟수
        int low = 0;           // 낮은 값
        int high = 0;          // 높은 값
        int mix = 0;           // 섞은 값

        // 배열 값 -> 큐에 저장
        for(int sc : scoville) {
            queue.offer(sc);
        }

        // queue의 사이즈가 2보다 작거나 나 최소 값이 K보다 크다면 바로 통과
        while(queue.size() >= 2 && queue.peek() < K) {
            // 큐에서 값을 2개 빼서 섞고, 다시 큐에 넣음
            low = queue.poll();
            high = queue.poll();
            
            mix = low + high*2;
            queue.offer(mix);
            count++;
        }

        /*
          while문에 들어가지 않을 경우, 조건 검사해서 최소 값이 K보다 작다면
          조합할 수 없기 때문에 return -1
        */
        if(queue.peek() < K) {
            return -1;
        }
        return count;
    }
}
```
