## 가장 가까운 같은 글자 No.142086

```java
import java.util.*;

class Solution {
    public int[] solution(String s) {
        int[] alphabet = new int[26];           // 알파벳을 인덱스 -> 값으로 맵처럼 구현
        Arrays.fill(alphabet, -1);              // 기본 값이 -1이므로 -1로 세팅
        char[] word = s.toCharArray();          // 인자 s를 char 배열로 저장
        int[] arr = new int[word.length];       // 정답을 저장할 배열 arr
        
        
        for(int i=0; i<word.length; i++) {
            char w = word[i];
            /* alphabet[w-'97'] 값이 -1이 아니라면 이전에 나왔으므로
                현재 인덱스 i - alphabet[w-'97'] 을 arr에 저장
                alphbet 배열에 값을 업데이트
            */
            if(alphabet[w-'a'] != -1) {         
                arr[i] = i - alphabet[w-'a'];
            }   else {
                arr[i] = -1;
            }
            alphabet[w-'a'] = i;
        }
        
        return arr;
    }
}
```
