
## 콜라 문제

class Solution {
     public int solution(int a, int b, int n) {
        if (n < a) return 0;

        int exchanged = (n / a) * b;    
        int next = exchanged + (n % a); 

        return exchanged + solution(a, b, next);
    }
}

## 큰 수 만들기

class Solution {
    public String solution(String number, int k) {
        String current = number;

        for (int i = 0; i < k; i++) {
            String best = "";

            for (int j = 0; j < current.length(); j++) {
                String candidate = current.substring(0, j) + current.substring(j + 1);

                if (best.equals("") || candidate.compareTo(best) > 0) {
                    best = candidate;
                }
            }

            current = best;
        }

        return current;
    }
}

## 단속카메라

import java.util.*;

class Solution {
    public int solution(int[][] routes) {
        int camera = 1;
        Arrays.sort(routes, (a, b) -> a[1] - b[1]);
        
        int position = routes[0][1];
        
        for(int i=1; i<routes.length; i++) {
            if(routes[i][0] > position) {
                camera++;
                position = routes[i][1];
            }
        }
        
        return camera;
    }
}
