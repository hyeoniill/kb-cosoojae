## 피로도

```java
import java.util.*;

class Solution {
    int max = 0;
    
    public int solution(int k, int[][] dungeons) {
        List<Node> list = new ArrayList<>();
        //던전을 리스트에 node로 추가
        for(int i=0; i<dungeons.length; i++) {
            list.add(new Node(dungeons[i][0], dungeons[i][1]));   
        }
        
        
        //각 노드에서 시작하여 모든 경우의 수를 찾음
        for(int i=0; i<list.size(); i++) {
            visit(list.get(i), list, k, 0);
        }
        
        
        
        return max;
    }
    //현재 노드, 노드 리스트, 현재 남은 피로도, 방문 던전 수
    public void visit(Node currentNode, List<Node> list, int currentFP, int count) {
        //최소 필요 피로도보다 현재 피로도가 적으면 count와 max중 큰 값 리턴
        if(currentFP < currentNode.minimumFP) {
            max = Math.max(max, count);
            return;
        }
        //피로도를 줄이고, 방문 함, count++
        currentFP -= currentNode.FP;
        currentNode.visited = true;
        count++;
        
        max = Math.max(max, count);
        
        //나머지 노드도 전부 순회
        for(Node n : list) {
            if(!n.visited) {
                visit(n, list, currentFP, count);
            }
        }
        
        //노드 순회가 끝나면 현재 노드의 방문을 초기화 해야 함
        currentNode.visited = false;
    }    
}

class Node {
    int FP = 0;
    int minimumFP = 0;
    boolean visited = false;

    public Node(int minimumFP, int FP) {
        this.FP = FP;
        this.minimumFP = minimumFP;
    }
}
```
