## 키패드

```java
import java.util.*;
class Solution {
    public String solution(int[] numbers, String hand) {
        int[][] arr = {{3,1},
                       {0,0},{0,1},{0,2},
                       {1,0},{1,1},{1,2},
                       {2,0},{2,1},{2,2},
                       {3,0},{3,2}};
        
        int currentLeft = 10;
        int currentRight = 11;
        
        StringBuilder sb = new StringBuilder();
        
        for(int number: numbers) {
            if(number == 1 || number == 4 || number == 7) {
                sb.append("L");
                currentLeft = number;
            } else if(number == 3 || number == 6 || number == 9) {
                sb.append("R");
                currentRight = number;
            } else {
                int leftDistance = Math.abs(arr[currentLeft][0]-arr[number][0])+Math.abs(arr[currentLeft][1]-arr[number][1]);
                int rightDistance = Math.abs(arr[currentRight][0]-arr[number][0])+Math.abs(arr[currentRight][1]-arr[number][1]);
                if(leftDistance == rightDistance) {
                    if(hand.equals("right")) {
                        currentRight = number;
                        sb.append("R");
                    } else {
                         currentLeft = number;
                        sb.append("L");
                    }
                } else if(leftDistance < rightDistance) {
                    sb.append("L");
                    currentLeft = number;
                } else {
                    sb.append("R");
                    currentRight = number;
                }
            }
        }
        
        
        return sb.toString();
    }
}
```

## 피로도

```java
import java.util.*;

class Solution {
    int max = 0;
    
    public int solution(int k, int[][] dungeons) {
        List<Node> list = new ArrayList<>();
        //던전을 리스트에 node로 추가
        for(int i=0; i<dungeons.length; i++) {
            list.add(new Node(dungeons[i][0], dungeons[i][1]));   
        }
        
        
        //각 노드에서 시작하여 모든 경우의 수를 찾음
        for(int i=0; i<list.size(); i++) {
            visit(list.get(i), list, k, 0);
        }
        
        
        
        return max;
    }
    //현재 노드, 노드 리스트, 현재 남은 피로도, 방문 던전 수
    public void visit(Node currentNode, List<Node> list, int currentFP, int count) {
        //최소 필요 피로도보다 현재 피로도가 적으면 count와 max중 큰 값 리턴
        if(currentFP < currentNode.minimumFP) {
            max = Math.max(max, count);
            return;
        }
        //피로도를 줄이고, 방문 함, count++
        currentFP -= currentNode.FP;
        currentNode.visited = true;
        count++;
        
        max = Math.max(max, count);
        
        //나머지 노드도 전부 순회
        for(Node n : list) {
            if(!n.visited) {
                visit(n, list, currentFP, count);
            }
        }
        
        //노드 순회가 끝나면 현재 노드의 방문을 초기화 해야 함
        currentNode.visited = false;
    }    
}

class Node {
    int FP = 0;
    int minimumFP = 0;
    boolean visited = false;

    public Node(int minimumFP, int FP) {
        this.FP = FP;
        this.minimumFP = minimumFP;
    }
}
```

## 정수 삼각형
```java
import java.io.*;
import java.util.*;

class Solution {
    //트리
    static int[][] tree;
    //메모이제이션
    static int[][] treeSum;
    
    public int solution(int[][] triangle) {
        tree = triangle;
        treeSum = new int[tree.length][];
        //treeSum 초기화
        for(int i=0; i<tree.length; i++) {
            treeSum[i] = new int[i+1];
            Arrays.fill(treeSum[i], -1);
        }
        //(0,0)번부터 시작
        return calc(0,0);
    }
    
    public int calc(int degree, int currentIndex) {
        //깊이가 tree.length와 같아지면 0
        if(degree == tree.length) {
            return 0;
        }
        
        int current = tree[degree][currentIndex];
        int Calculated = treeSum[degree][currentIndex];
        
        //이미 계산한 값이면 계산한 값을 가져옴
        if(Calculated != -1) {
            return Calculated;
        }
        
        int currentMax = Math.max(current + calc(degree+1, currentIndex), current + calc(degree+1, currentIndex+1));
        treeSum[degree][currentIndex] = currentMax;
        
        return currentMax;
    }
}
```
